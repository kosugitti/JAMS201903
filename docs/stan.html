Stanを使った実習

7scientist

データを入力します。

## パッケージの読み込み
library(rstan)
library(tidyverse)
## データの準備
X <- c(-27.020,3.570,8.191,9.808,9.603,9.945,10.056)
sc7 <- list(N=NROW(X),X=X)

モデルをコンパイル

## モデルコンパイル
model <- stan_model("SevenScientist.stan")

コンパイルするためのソースコード

## data{
##   int<lower=0> N;
##   real X[N];
## }
## 
## parameters{
##   real mu;
##   real<lower=0> sig[N];
## }
## 
## model{
##   for(n in 1:N){
##     //likelihood
##     X[n] ~ normal(mu,sig[n]);
##     //prior
##     sig[n] ~ inv_gamma(0.0001,0.0001);
##   }
##   // prior
##   mu ~ normal(0,1000);
## }

推定

fit <- sampling(model,sc7,iter=3000,warmup=1000,chains=4,thin=1)
